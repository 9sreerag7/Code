#include stdio.h
void insertionSort(int a[], int n){
for (int i = 1; i < n; i++) {
int key = a[i];
int j = i - 1;
while (j >= 0 && a[j] > key) {
a[j + 1] = a[j];
j = j - 1;
}
a[j + 1] = key;
}
}

void printArray(int a[], int n) {
for (int i = 0; i < n; i++) {
printf("%d ", a[i]);
}
printf("\n");
}
int main() {
int a[50], n;
printf("Enter the number of elements: ")
; scanf("%d", &n);
printf("Enter the elements: ");
for (int i = 0; i < n; i++) {
scanf("%d", &a[i]);
}
insertionSort(a, n);
printf("Sorted array: ");
printArray(a, n);
return 0;
}
